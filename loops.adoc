== Loops

:file-name: loops.clj

include::use_code_note.adoc[]

=== for

(let [nums [1 2 3 4]]
  (for [num nums]
       (* num 10)))

(def colors ["red" "blue" "green" "yellow"])

(def shapes ["square" "circle" "triangle" "rectangle"])

(for [color colors
      shape shapes]
  (str color " " shape))

=== doseq

(let [nums [1 2 3 4]]
  (doseq [num nums]
    (* num 10)))

(let [nums [1 2 3 4]]
  (doseq [num nums]
    (println (* num 10))))

=== loop

(loop [x 1]
  (when (<= x 5)
    (println x)
    (recur (inc x))))
