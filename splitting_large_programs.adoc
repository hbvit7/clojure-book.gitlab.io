== Splitting Large Programs

[source, clojure]
----
;; calculator.clj

(defn add [a b]
  (+ a b))

(defn sub [a b]
  (- a b))

(def a 5)
(def b 3)

(println a " + " b " = " (add a b))
(println a " - " b " = " (sub a b))
----

Output

----
5  +  3  =  8
5  -  3  =  2
----
