=== apply


[source, clojure]
----
(+ 1 2 3 4)
----


[source, clojure]
----
(+ [1 2 3 4]) ;; error
----


[source, clojure]
----
(apply + [1 2 3 4])
----


[source, clojure]
----
(apply + '(1 2 3 4))
----

[source, clojure]
----
(defn my-function [arg1 arg2]
  (println "Argument 1:" arg1)
  (println "Argument 2:" arg2))
----

[source, clojure]
----
(my-function "one" "two")
----


[source, clojure]
----
(my-function ["one" "two"]) ;; error
----


[source, clojure]
----
(apply my-function ["one" "two"])
----
